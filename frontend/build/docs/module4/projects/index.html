Module 4 Projects

In this section, you'll work on comprehensive projects that integrate all four modules to create complete Physical AI systems capable of understanding natural language commands, perceiving their environment, and executing complex tasks.

Project 1: Complete VLA Home Assistant Robot​

Objective​

Create a complete VLA system that acts as a home assistant robot, responding to voice commands to perform household tasks.

Requirements​

  • Implement voice command recognition using Whisper
  • Integrate with LLM for natural language understanding
  • Use perception system to identify objects and navigate
  • Execute complex multi-step tasks safely
  • Provide feedback to users through text-to-speech
  • Integrate with all previous modules (ROS 2, Digital Twin, AI-Robot Brain)

Implementation Steps​

    def voice_command_callback(self, msg):
        """Handle voice commands from user."""
        command_text = msg.data.strip()
        self.get_logger().info(f'Received voice command: "{command_text}"')

        # Validate command
        if self.is_command_allowed(command_text):
            # Create task
            task = HouseholdTask(
                id=f"task_{self.get_clock().now().nanoseconds}",
                type="command",
                target=command_text
            )

            # Add to queue
            self.task_queue.put(task)
            self.get_logger().info('Command added to task queue')
        else:
            self.provide_feedback("Command not allowed during current conditions")

    def interpreted_command_callback(self, msg):
        """Handle interpreted commands from LLM."""
        interpreted_cmd = msg.data
        self.get_logger().info(f'Interpreted command: {interpreted_cmd}')

        if interpreted_cmd.startswith('FETCH_ITEM:'):
            item = interpreted_cmd.split(':')[1]
            self.create_fetch_task(item)
        elif interpreted_cmd.startswith('NAVIGATE_TO:'):
            location = interpreted_cmd.split(':')[1]
            self.create_navigation_task(location)
        elif interpreted_cmd.startswith('FIND_ITEM:'):
            item = interpreted_cmd.split(':')[1]
            self.create_find_task(item)
        elif interpreted_cmd.startswith('BRING_ITEM:'):
            parts = interpreted_cmd.split(':')
            item = parts[1]
            destination = parts[2] if len(parts) > 2 else "current_location"
            self.create_bring_task(item, destination)
        elif interpreted_cmd.startswith('MONITOR_AREA:'):
            area = interpreted_cmd.split(':')[1]
            self.create_monitoring_task(area)

    def perception_callback(self, msg):
        """Handle perception data."""
        self.detected_objects = []
        for detection in msg.detections:
            if detection.results:
                for result in detection.results:
                    if result.hypothesis.score > 0.5:  # Confidence threshold
                        self.detected_objects.append({
                            'label': result.hypothesis.class_id,
                            'confidence': result.hypothesis.score,
                            'position': detection.bbox.center
                        })

    def odom_callback(self, msg):
        """Update robot pose."""
        pose = msg.pose.pose
        self.robot_pose["x"] = pose.position.x
        self.robot_pose["y"] = pose.position.y

        # Convert orientation to angle
        from tf_transformations import euler_from_quaternion
        import math
        (_, _, theta) = euler_from_quaternion([
            pose.orientation.x,
            pose.orientation.y,
            pose.orientation.z,
            pose.orientation.w
        ])
        self.robot_pose["theta"] = theta

    def scan_callback(self, msg):
        """Handle laser scan for obstacle detection."""
        # Check for obstacles in path
        min_distance = min([r for r in msg.ranges if r > 0 and r < float('inf')], default=float('inf'))

        if min_distance < 0.5:  # Too close to obstacle
            self.activate_safety_protocol("OBSTACLE_TOO_CLOSE")

    def action_feedback_callback(self, msg):
        """Handle action execution feedback."""
        feedback = msg.data
        self.get_logger().info(f'Action feedback: {feedback}')

        if feedback.startswith("SUCCESS:"):
            if self.active_task:
                self.complete_task("success")
        elif feedback.startswith("FAILURE:") or feedback.startswith("ERROR:"):
            if self.active_task:
                self.complete_task("failure")

    def main_loop(self):
        """Main system loop."""
        if self.current_state == TaskState.IDLE and not self.task_queue.empty():
            self.start_next_task()
        elif self.current_state == TaskState.EXECUTING:
            self.continue_task_execution()
        elif self.current_state == TaskState.SAFETY_STOP:
            self.handle_safety_stop()

    def start_next_task(self):
        """Start the next task in the queue."""
        if not self.task_queue.empty():
            try:
                task = self.task_queue.get_nowait()
                self.active_task = task
                self.current_state = TaskState.PROCESSING

                self.get_logger().info(f'Starting task: {task.type} - {task.target}')

                # Process the task
                self.execute_household_task(task)

            except queue.Empty:
                pass

    def execute_household_task(self, task):
        """Execute a household task."""
        if task.type == "fetch_item":
            self.execute_fetch_task(task)
        elif task.type == "navigation":
            self.execute_navigation_task(task)
        elif task.type == "find_item":
            self.execute_find_task(task)
        elif task.type == "bring_item":
            self.execute_bring_task(task)
        elif task.type == "monitor_area":
            self.execute_monitoring_task(task)
        else:
            self.get_logger().warn(f'Unknown task type: {task.type}')
            self.complete_task("failure")

    def execute_fetch_task(self, task):
        """Execute fetch item task."""
        item = task.target
        self.get_logger().info(f'Fetching item: {item}')

        # Find item location
        item_location = self.find_item_location(item)
        if item_location:
            # Navigate to item location
            self.navigate_to_location(item_location)

            # Wait for navigation to complete, then grasp
            # This would be handled by a state machine
            self.current_state = TaskState.EXECUTING
        else:
            self.get_logger().warn(f'Item {item} location unknown')
            self.provide_feedback(f"Sorry, I don't know where the {item} is located")
            self.complete_task("failure")

    def execute_navigation_task(self, task):
        """Execute navigation task."""
        location = task.target
        if location in self.household_map["rooms"]:
            self.navigate_to_location(location)
            self.current_state = TaskState.EXECUTING
        else:
            self.get_logger().warn(f'Unknown location: {location}')
            self.provide_feedback(f"Sorry, I don't know where {location} is")
            self.complete_task("failure")

    def execute_find_task(self, task):
        """Execute find item task."""
        item = task.target
        self.get_logger().info(f'Finding item: {item}')

        # Search for item in known locations
        if item in self.household_map["objects"]:

    def execute_bring_task(self, task):
        """Execute bring item task."""
        item = task.target
        destination = task.destination or "current_location"

        self.get_logger().info(f'Bringing {item} to {destination}')

        # Complex task: fetch + navigate + deliver
        self.multi_step_task = [
            ("find", item),
            ("fetch", item),
            ("navigate", destination),
            ("deliver", item)
        ]
        self.current_task_step = 0
        self.execute_task_step()

    def execute_task_step(self):
        """Execute the current step in a multi-step task."""
        if hasattr(self, 'multi_step_task') and self.current_task_step < len(self.multi_step_task):
            action, target = self.multi_step_task[self.current_task_step]

            if action == "find":
                self.execute_find_task(HouseholdTask("temp", "find_item", target))
            elif action == "fetch":
                self.execute_fetch_task(HouseholdTask("temp", "fetch_item", target))
            elif action == "navigate":
                self.execute_navigation_task(HouseholdTask("temp", "navigation", target))
            elif action == "deliver":
                # Place item at current location
                cmd_msg = String()
                cmd_msg.data = f"PLACE:{target}"
                self.action_command_pub.publish(cmd_msg)

            self.current_task_step += 1
